28  package net.chwthewke.hamcrest.annotations;
29  
30  import java.lang.annotation.ElementType;
31  import java.lang.annotation.Retention;
32  import java.lang.annotation.RetentionPolicy;
33  import java.lang.annotation.Target;
34  
35  /**
36   * The {@link EquivalenceSpecificationOn} annotation declares that the annotated type (which <strong>must</strong> be a <strong>public</strong> interface) is
37   * an <em>equivalence specification</em> for the type returned by {@link #value()} (henceforth called <em>matched type</em>).
38   * <p>
39   * This imposes further restrictions on the interface, namely that each of its methods must:
40   * <ul>
41   * <li>have a non-<code>void</code> return type ({@link Void} is ok, though),</li>
42   * <li>have no arguments</li>
43   * <li>have the same signature as a <strong>public</strong> method of the matched type, with the following derogations:
44   * <ul>
45   * <li>if the interface method is annotated with {@link NotPublic}, the method must simply be visible from the matched type (i.e. it can be
46   * declared on a superclass if protected, etc.)</li>
47   * <li>covariant overriding is supported, i.e. the return type of the method on the interface may be a superclass of the return type of the
48   * method on the matched type.
49   * </ul>
50   * </li>
51   * </ul>
52   * <p>
53   * Each of the methods of the interface may also be annotated with at most one annotation used to define equivalence for that property
54   * method. Placing more than one such annotation on a method is unsupported and may result in an error in a later version. The available
55   * equivalence annotations are:
56   * <ul>
57   * <li>{@link Equality},</li>
58   * <li>{@link Identity},</li>
59   * <li>{@link ApproximateEquality},</li>
60   * <li>{@link BySpecification}</li>
61   * </ul>
62   * If no equivalence annotation is present on a method, it is treated as if it bore the {@link Equality} annotation.
63   */
64  @Target( ElementType.TYPE )
65  @Retention( RetentionPolicy.RUNTIME )
66  public @interface EquivalenceSpecificationOn {
67      /**
68       * The type this equivalence specification acts upon.
69       */
70      Class<?> value( );
71  }
